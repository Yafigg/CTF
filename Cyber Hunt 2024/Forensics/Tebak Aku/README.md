```Python
import zipfile
import itertools

# Open the wordlist file
with open('wordlist.txt', 'r') as f:
    wordlist = [line.strip() for line in f.readlines()]

# Open the ZIP archive with the password
with zipfile.ZipFile('congrats.zip', 'r') as zip_file:
    # Try each word in the wordlist as a password
    for word in wordlist:
        try:
            zip_file.setpassword(word.encode())
            zip_file.extractall()
            print(f"Password found: {word}")
            break
        except RuntimeError:
            pass
```
